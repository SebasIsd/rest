public class pedidosController {
    private static final String API_URL = "http://localhost/reptom/apiPedidos.php";
    private static final HttpClient client = HttpClient.newHttpClient();

    public static String listar() {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(API_URL))
                    .GET()
                    .build();
            return client.send(request, HttpResponse.BodyHandlers.ofString()).body();
        } catch (Exception e) {
            return "[]";
        }
    }

    public static String insertar(Pedido a) {
    try {
        String body = "idPedido=" + encode(a.getIdPedido(), "UTF-8")
                + "&idCliente=" + encode(a.getIdCliente(), "UTF-8")
                + "&producto=" + encode(a.getProducto(), "UTF-8");
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL))
                .header("Content-Type", "application/x-www-form-urlencoded")
                .POST(HttpRequest.BodyPublishers.ofString(body))
                .build();
        return client.send(request, HttpResponse.BodyHandlers.ofString()).body();
    } catch (Exception e) {
        return null;
    }
}

public static String eliminar(String id) {
    try {
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL + "?idPedido=" + encode(id, "UTF-8")))
                .DELETE()
                .build();
        return client.send(request, HttpResponse.BodyHandlers.ofString()).body();
    } catch (Exception ex) {
        return null;
    }
}

public static String actualizar(Pedido a) {
    try {
        String params = "?idPedido=" + encode(a.getIdPedido(), "UTF-8")
                + "&idCliente=" + encode(a.getIdCliente(), "UTF-8")
                + "&producto=" + encode(a.getProducto(), "UTF-8");
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL + params))
                .header("Content-Type", "application/x-www-form-urlencoded")
                .method("PUT", HttpRequest.BodyPublishers.noBody())
                .build();
        return client.send(request, HttpResponse.BodyHandlers.ofString()).body();
    } catch (Exception ex) {
        return null;
    }
}

public static String buscar(String nombre) {
    try {
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL + "?idPedido=" + encode(nombre, "UTF-8")))
                .GET()
                .build();
        return client.send(request, HttpResponse.BodyHandlers.ofString()).body();

@WebServlet(name = "ClientesServlet", urlPatterns = {"/clientes"})
public class ClientesServlet extends HttpServlet {
    private Gson gson = new Gson();

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        String accion = request.getParameter("accion");

        if (accion == null) {
            String json = clientesController.listar();
            Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> mapaLista = gson.fromJson(json, tipo);
            List<Cliente> lista = new ArrayList<>();
            for (Map<String, Object> m : mapaLista) {
                lista.add(new Cliente(
                    String.valueOf(m.get("cedula")),
                    String.valueOf(m.get("nombre")),
                    String.valueOf(m.get("apellido"))
                ));
            }
            request.setAttribute("clientes", lista);
            request.getRequestDispatcher("/listarClientes.jsp").forward(request, response);
        }
        else if ("buscar".equals(accion)) {
            String nombre = request.getParameter("nombre");
            String json = clientesController.buscar(nombre);
            Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> lista = gson.fromJson(json, tipo);
            request.setAttribute("clientes", lista);
            request.getRequestDispatcher("/listarClientes.jsp").forward(request, response);
        }
        else if ("irAgregar".equals(accion)) {
            request.getRequestDispatcher("/agregarCliente.jsp").forward(request, response);
        }
        else if ("editar".equals(accion)) {
            String id = request.getParameter("cedula");
            String json = clientesController.listar();
            Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> mapaLista = gson.fromJson(json, tipo);
            Cliente encontrado = null;
            for (Map<String, Object> m : mapaLista) {
                if (String.valueOf(m.get("cedula")).equals(id)) {
                    encontrado = new Cliente(
                        String.valueOf(m.get("cedula")),
                        String.valueOf(m.get("nombre")),
                        String.valueOf(m.get("apellido"))
                    );
                    break;
                }
            }
            request.setAttribute("cliente", encontrado);
            request.getRequestDispatcher("/editarCliente.jsp").forward(request, response);
        }
        else if ("eliminar".equals(accion)) {
            String id = request.getParameter("cedula");
            clientesController.eliminar(id);
            response.sendRedirect("clientes");
        }
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        String accion = request.getParameter("accion");

        if ("crear".equals(accion)) {
            Cliente c = new Cliente(
                request.getParameter("cedula"),
                request.getParameter("nombre"),
                request.getParameter("apellido")
            );
            clientesController.insertar(c);
        }
        else if ("actualizar".equals(accion)) {
            Cliente c = new Cliente(
                request.getParameter("cedula"),
                request.getParameter("nombre"),
                request.getParameter("apellido")
            );
            clientesController.actualizar(c);
        }
        response.sendRedirect("clientes");
    }
}

@WebServlet(name = "PedidosServlet", urlPatterns = {"/pedidos"})
public class PedidosServlet extends HttpServlet {

    private Gson gson = new Gson();

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        String accion = request.getParameter("accion");

        if (accion == null) {
            String json = pedidosController.listar();
            List<Pedido> lista = gson.fromJson(json,
                    new TypeToken<List<Pedido>>(){}.getType());
            request.setAttribute("pedidos", lista);
            request.getRequestDispatcher("/listarPedidos.jsp").forward(request, response);
        }
        else if ("buscar".equals(accion)) {
            String nombre = request.getParameter("nombre");
            String json = pedidosController.buscar(nombre);
            Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> lista = gson.fromJson(json, tipo);
            request.setAttribute("pedidos", lista);
            request.getRequestDispatcher("/listarPedidos.jsp").forward(request, response);
        }
        else if ("irAgregar".equals(accion)) {
            String jsonClientes = clientesController.listar();
            Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> mapaLista = gson.fromJson(jsonClientes, tipo);
            List<Cliente> clientesList = new ArrayList<>();
            for (Map<String, Object> m : mapaLista) {
                clientesList.add(new Cliente(
                    String.valueOf(m.get("cedula")),
                    String.valueOf(m.get("nombre")),
                    String.valueOf(m.get("apellido"))
                ));
            }
            request.setAttribute("clientes", clientesList);
            request.getRequestDispatcher("/agregarPedido.jsp").forward(request, response);
        }
        else if ("editar".equals(accion)) {
            String id = request.getParameter("idPedido");
            String json = pedidosController.listar();
            List<Pedido> mapaLista = gson.fromJson(json,
                    new TypeToken<List<Pedido>>(){}.getType());
            Pedido encontrado = null;
            for (Pedido p : mapaLista) {
                if (p.getIdPedido().equals(id)) {
                    encontrado = p;
                    break;
                }
            }
            String jsonClientes = clientesController.listar();
            Type tipoClientes = new TypeToken<List<Map<String, Object>>>(){}.getType();
            List<Map<String, Object>> mapaClientes = gson.fromJson(jsonClientes, tipoClientes);
            List<Cliente> clientesList = new ArrayList<>();
            for (Map<String, Object> m : mapaClientes) {
                clientesList.add(new Cliente(
                    String.valueOf(m.get("cedula")),
                    String.valueOf(m.get("nombre")),
                    String.valueOf(m.get("apellido"))
                ));
            }
            request.setAttribute("pedido", encontrado);
            request.setAttribute("clientes", clientesList);
            request.getRequestDispatcher("/editarPedido.jsp").forward(request, response);
        }
        else if ("eliminar".equals(accion)) {
            String id = request.getParameter("idPedido");
            pedidosController.eliminar(id);
            response.sendRedirect("pedidos");
        }
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        String accion = request.getParameter("accion");

        if ("crear".equals(accion)) {
            Pedido p = new Pedido(
                request.getParameter("idPedido"),
                request.getParameter("cedula"),
                request.getParameter("producto")
            );
            pedidosController.insertar(p);
        }
        else if ("actualizar".equals(accion)) {
            Pedido p = new Pedido(
                request.getParameter("idPedido"),
                request.getParameter("cedula"),
                request.getParameter("producto")
            );
            pedidosController.actualizar(p);
        }
        response.sendRedirect("pedidos");
    }

$conexion=$con->conectar();

$listar="select * from clientes";
$resultado= $conexion->prepare($listar);
$resultado->execute();
$data=$resultado->fetchAll(PDO::FETCH_ASSOC);
echo json_encode($data);
}

public static function insertar(){
$con= new conexion();
$conexion=$con->conectar();

$cedula=$_POST['cedula'];
$nombre=$_POST['nombre'];
$apellido=$_POST['apellido'];

$insertar="insert into clientes values ('$cedula','$nombre','$apellido')";
$resultado=$conexion->prepare($insertar);
$resultado->execute();
echo json_encode(["se inserto"]);
}
public static function actualizar(){
$con= new conexion();
$conexion=$con->conectar();
$cedula=$_GET['cedula'];
$nombre=$_GET['nombre'];
$apellido=$_GET['apellido'];

$actualizar="update clientes set nombre='$nombre',apellido='$apellido' where cedula='$cedula'";
$resultado=$conexion->prepare($actualizar);
$resultado->execute();
echo json_encode(["se actualizo"]);
}
public static function buscar(){
$con= new conexion();
$conexion=$con->conectar();
 $nombre=$_GET['nombre'];
$buscar="select c.cedula,c.nombre,c.apellido,p.idPedido,
                p.idCliente,p.producto from clientes c
                inner join pedidos p on c.cedula=p.idCliente
                where c.nombre='$nombre'";
$resultado= $conexion->prepare($buscar);
$resultado->execute();
$data=$resultado->fetchAll(PDO::FETCH_ASSOC);
echo json_encode($data);
}
