package com.uta.repaso.entities;

import jakarta.persistence.*;
import java.util.List;

@Entity
@Table(name = "autores")
public class Autor {

    @Id
    @Column(name = "idAutor")
    private String idAutor;

    @Column(name = "nombre")
    private String nombre;

    @Column(name = "apellido")
    private String apellido;

    @Column(name = "correo")
    private String correo;

    @OneToMany(mappedBy = "autor", cascade = CascadeType.ALL)
    private List<Libro> libros;

    public Autor() {}

    public String getIdAutor() { return idAutor; }
    public void setIdAutor(String idAutor) { this.idAutor = idAutor; }

    public String getNombre() { return nombre; }
    public void setNombre(String nombre) { this.nombre = nombre; }

    public String getApellido() { return apellido; }
    public void setApellido(String apellido) { this.apellido = apellido; }

    public String getCorreo() { return correo; }
    public void setCorreo(String correo) { this.correo = correo; }

    public List<Libro> getLibros() { return libros; }
    public void setLibros(List<Libro> libros) { this.libros = libros; }
}


package com.uta.repaso.entities;

import jakarta.persistence.*;

@Entity
@Table(name = "libros")
public class Libro {

    @Id
    @Column(name = "idLibro")
    private String idLibro;

    @Column(name = "titulo")
    private String titulo;

    @Column(name = "anio_publicacion")
    private Integer anio_publicacion;

    @ManyToOne
    @JoinColumn(name = "id_autor")
    private Autor autor;

    public Libro() {}

    public String getIdLibro() { return idLibro; }
    public void setIdLibro(String idLibro) { this.idLibro = idLibro; }

    public String getTitulo() { return titulo; }
    public void setTitulo(String titulo) { this.titulo = titulo; }

    public Integer getAnio_publicacion() { return anio_publicacion; }
    public void setAnio_publicacion(Integer anio_publicacion) { this.anio_publicacion = anio_publicacion; }

    public Autor getAutor() { return autor; }
    public void setAutor(Autor autor) { this.autor = autor; }
}

package com.uta.repaso.servicios;

import com.uta.repaso.entities.Autor;
import com.uta.repaso.repository.AutorRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.util.List;

@Service
public class AutorServiceImpl implements AutorService {

    @Autowired
    private AutorRepository repository;

    @Override
    public List<Autor> listar() {
        return repository.findAll();
    }

    @Override
    public void guardar(Autor autor) {
        repository.save(autor);
    }

    @Override
    public void eliminar(String idAutor) {
        repository.deleteById(idAutor);
    }

    @Override
    public List<Autor> buscarPorNombre(String nombre) {
        return repository.findByNombre(nombre);
    }

    @Override
    public Autor buscarPorId(String idAutor) {
        return repository.findById(idAutor).orElse(null);
    }
}

package com.uta.repaso.controller;

import com.uta.repaso.entities.Autor;
import com.uta.repaso.servicios.AutorService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import java.util.List;

@Controller
@RequestMapping("/autores")
public class AutorController {

    @Autowired
    private AutorService autorService;

    @GetMapping
    public String listar(Model model) {
        model.addAttribute("autor", new Autor());
        model.addAttribute("autores", autorService.listar());
        return "autores";
    }

    @PostMapping("/guardar")
    public String guardar(Autor autor) {
        autorService.guardar(autor);
        return "redirect:/autores";
    }

    @GetMapping("/buscar")
    public String buscar(@RequestParam String nombre, Model model) {
        List<Autor> resultado = autorService.buscarPorNombre(nombre);
        model.addAttribute("autor", new Autor());
        model.addAttribute("autores", autorService.listar());
        if (resultado != null && !resultado.isEmpty()) {
            model.addAttribute("autorEncontrado", resultado.get(0));
            model.addAttribute("librosAutor", resultado.get(0).getLibros());
        }
        return "autores";
    }

    @GetMapping("/editar/{id}")
    public String editar(@PathVariable String id, Model model) {
        model.addAttribute("autor", autorService.buscarPorId(id)); // carga datos en el form
        model.addAttribute("autores", autorService.listar());
        return "autores";
    }

    @GetMapping("/eliminar/{id}")
    public String eliminar(@PathVariable String id) {
        autorService.eliminar(id);
        return "redirect:/autores";
    }
}

package com.uta.repaso.controller;

import com.uta.repaso.entities.Autor;
import com.uta.repaso.entities.Libro;
import com.uta.repaso.servicios.AutorService;
import com.uta.repaso.servicios.LibroService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import java.util.List;

@Controller
@RequestMapping("/libros")
public class LibroController {

    @Autowired
    private LibroService libroService;

    @Autowired
    private AutorService autorService;

    @GetMapping
    public String listar(Model model) {
        model.addAttribute("libro", new Libro());
        model.addAttribute("libros", libroService.listar());
        model.addAttribute("autores", autorService.listar());
        return "libros";
    }

    @PostMapping("/guardar")
    public String guardar(
            @RequestParam String idLibro,
            @RequestParam String titulo,
            @RequestParam Integer anio_publicacion,
            @RequestParam String idAutor) {

        Autor autor = autorService.buscarPorId(idAutor);
        Libro libro = new Libro();
        libro.setIdLibro(idLibro);
        libro.setTitulo(titulo);
        libro.setAnio_publicacion(anio_publicacion);
        libro.setAutor(autor);
        libroService.guardar(libro);
        return "redirect:/libros";
    }

    @GetMapping("/buscar")
    public String buscar(@RequestParam String titulo, Model model) {
        List<Libro> resultado = libroService.buscarPorTitulo(titulo);
        model.addAttribute("libro", new Libro());
        model.addAttribute("libros", libroService.listar());
        model.addAttribute("autores", autorService.listar());
        if (resultado != null && !resultado.isEmpty()) {
            model.addAttribute("libroEncontrado", resultado.get(0));
        }
        return "libros";
    }

    @GetMapping("/editar/{id}")
    public String editar(@PathVariable String id, Model model) {
        Libro libro = libroService.buscarPorId(id); // carga datos en el form
        model.addAttribute("libro", libro);
        model.addAttribute("libros", libroService.listar());
        model.addAttribute("autores", autorService.listar());
        return "libros";
    }

    @GetMapping("/eliminar/{id}")
    public String eliminar(@PathVariable String id) {
        libroService.eliminar(id);
        return "redirect:/libros";
    }
}
