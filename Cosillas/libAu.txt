@WebServlet(name = "LibroServlet", urlPatterns = {"/libros"})
public class LibroServlet extends HttpServlet {

    private Gson gson = new Gson();

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String accion = request.getParameter("accion");

        
        if (accion == null) {

            String json = librosController.listar();

            List<libros> lista = gson.fromJson(json,
                    new TypeToken<List<libros>>(){}.getType());

            request.setAttribute("libros", lista);
            request.getRequestDispatcher("/listarLibros.jsp").forward(request, response);
        }

        else if ("buscar".equals(accion)) {

            String titulo = request.getParameter("titulo");
            String json = librosController.buscar(titulo);

             Type tipo = new TypeToken<List<Map<String, Object>>>() {
            }.getType();
            List<Map<String, Object>> lista = gson.fromJson(json, tipo);

            request.setAttribute("libros", lista);
            request.getRequestDispatcher("/listarLibros.jsp").forward(request, response);
        }

       else if ("irAgregar".equals(accion)) {

    String json = autoresController.listar();
    List<autores> autoresList = gson.fromJson(json,
            new TypeToken<List<autores>>(){}.getType());
    request.setAttribute("autores", autoresList);
    request.getRequestDispatcher("/agregarLibro.jsp").forward(request, response);
}

      else if ("editar".equals(accion)) {
    String id = request.getParameter("id_libro");

    String json = librosController.listar();
    Type tipo = new TypeToken<List<Map<String, Object>>>(){}.getType();
    List<Map<String, Object>> mapaLista = gson.fromJson(json, tipo);
    libros encontrado = null;
    for (Map<String, Object> m : mapaLista) {
        if (String.valueOf(m.get("id_libro")).equals(id)) {
            encontrado = new libros(
                String.valueOf(m.get("id_libro")),
                String.valueOf(m.get("titulo")),
                String.valueOf(m.get("anio_publicacion")),
                String.valueOf(m.get("id_autor"))
            );
            break;
        }
    }

   
    String jsonAutores = autoresController.listar();
    List<autores> autoresList = gson.fromJson(jsonAutores,
            new TypeToken<List<autores>>(){}.getType());

    request.setAttribute("libro", encontrado);
    request.setAttribute("autores", autoresList);
    request.getRequestDispatcher("/editarLibro.jsp").forward(request, response);
}

        // ELIMINAR
        else if ("eliminar".equals(accion)) {

            String id = request.getParameter("id_libro");
            librosController.eliminar(id);

            response.sendRedirect("libros");
        }
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String accion = request.getParameter("accion");

        // CREAR
        if ("crear".equals(accion)) {

            libros l = new libros(
                    request.getParameter("id_libro"),
                    request.getParameter("titulo"),
                    request.getParameter("anio_publicacion"),
                    request.getParameter("id_autor")
            );

            librosController.insertar(l);
        }

        // ACTUALIZAR
        else if ("actualizar".equals(accion)) {

            libros l = new libros(
                    request.getParameter("id_libro"),
                    request.getParameter("titulo"),
                    request.getParameter("anio_publicacion"),
                    request.getParameter("id_autor")
            );

            librosController.actualizar(l);
        }

        response.sendRedirect("libros");
    }
}



@WebServlet(name = "AutoresServlet", urlPatterns = {"/autores"})
public class AutoresServlet extends HttpServlet {

    private Gson gson = new Gson();

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String accion = request.getParameter("accion");

        // LISTAR
        if (accion == null) {

            String json = autoresController.listar();

            List<autores> lista = gson.fromJson(json,
                    new TypeToken<List<autores>>() {
                    }.getType());

            request.setAttribute("autores", lista);
            request.getRequestDispatcher("/listarAutores.jsp").forward(request, response);
        } // BUSCAR
        else if ("buscar".equals(accion)) {

            String nombre = request.getParameter("nombre");
            String json = autoresController.buscar(nombre);

            Type tipo = new TypeToken<List<Map<String, Object>>>() {
            }.getType();
            List<Map<String, Object>> lista = gson.fromJson(json, tipo);

            request.setAttribute("autores", lista);
            request.getRequestDispatcher("/listarAutores.jsp").forward(request, response);
        } // IR AGREGAR
        else if ("irAgregar".equals(accion)) {
            request.getRequestDispatcher("/agregarAutor.jsp").forward(request, response);
        } // EDITAR
        else if ("editar".equals(accion)) {

            String id = request.getParameter("id_Autor");
            String json = autoresController.listar();

            List<autores> lista = gson.fromJson(json,
                    new TypeToken<List<autores>>() {
                    }.getType());

            autores encontrado = null;

            for (autores a : lista) {
                if (a.getId_Autor().equals(id)) {
                    encontrado = a;
                    break;
                }
            }

            request.setAttribute("autor", encontrado);
            request.getRequestDispatcher("/editarAutor.jsp").forward(request, response);
        } // ELIMINAR
        else if ("eliminar".equals(accion)) {

            String id = request.getParameter("id_Autor");
            autoresController.eliminar(id);

            response.sendRedirect("autores");
            return;
        }
    }

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String accion = request.getParameter("accion");

        // CREAR
        if ("crear".equals(accion)) {

            autores a = new autores(
                    request.getParameter("id_Autor"),
                    request.getParameter("nombre"),
                    request.getParameter("apellido"),
                    request.getParameter("correo")
            );

            autoresController.insertar(a);
        } // ACTUALIZAR
        else if ("actualizar".equals(accion)) {

            autores a = new autores(
                    request.getParameter("id_Autor"),
                    request.getParameter("nombre"),
                    request.getParameter("apellido"),
                    request.getParameter("correo")
            );

            autoresController.actualizar(a);
        }

        response.sendRedirect(request.getContextPath() + "/autores");
    }
}


public class librosController {
    private static final String API_URL="http://localhost/reptom/apiLibros.php";
    private static final HttpClient client = HttpClient.newHttpClient();
    
public static String listar() {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(API_URL))
                    .GET()
                    .build();

            return client.send(request, HttpResponse.BodyHandlers.ofString()).body();

        } catch (Exception e) {
            return "[]";
        }
    }

    public static String insertar(libros l) {
        try {
            String body = "id_libro=" + encode(l.getId_libro())
                    + "&titulo=" + encode(l.getTitulo())
                    + "&anio_publicacion=" + encode(l.getAnio_publicacion())
                    + "&id_autor=" + encode(l.getId_autor());

            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(API_URL))
                    .header("Content-Type", "application/x-www-form-urlencoded")
                    .POST(HttpRequest.BodyPublishers.ofString(body))
                    .build();

            return client.send(request, HttpResponse.BodyHandlers.ofString()).body();

        } catch (Exception e) {
            return null;
        }
    }

    public static String eliminar(String id) {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(API_URL + "?id_libro=" + encode(id)))
                    .DELETE()
                    .build();

            return client.send(request, HttpResponse.BodyHandlers.ofString()).body();

        } catch (Exception e) {
            return null;
        }
    }
     public static String buscar(String titulo) {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(API_URL + "?titulo=" + encode(titulo)))
                    .GET()
                    .build();

            return client.send(request, HttpResponse.BodyHandlers.ofString()).body();

        } catch (Exception e) {
            return "[]";
        }
    }
  public static String actualizar(libros l) {
    try {
        String params = "?id_libro=" + encode(l.getId_libro(), "UTF-8")
                + "&titulo=" + encode(l.getTitulo(), "UTF-8")
                + "&anio_publicacion=" + encode(l.getAnio_publicacion(), "UTF-8")
                + "&id_autor=" + encode(l.getId_autor(), "UTF-8");

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL + params))
                .header("Content-Type", "application/x-www-form-urlencoded")
                .method("PUT", HttpRequest.BodyPublishers.noBody())
                .build();
        return client.send(request, HttpResponse.BodyHandlers.ofString()).body();
    } catch (Exception e) {
        System.out.println("Error ACTUALIZAR libro: " + e.getMessage());
        return null;
    }
}
}
