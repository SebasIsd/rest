import java.io.BufferedReader;

public class api {

    private static final String URL_AUTOR = "http://localhost/SERVICIOS_MVC/controller/api.php";
    private static final String URL_LIBRO = "http://localhost/SERVICIOS_MVC/controller/apiLibros.php";

    public static String getAutores() {
        try {
            URL url = new URL(URL_AUTOR);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream()));
            StringBuilder response = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
            reader.close();
            return response.toString();
        } catch (Exception ex) {
            return null;
        }
    }

    public static String insertarAutor(String id, String nombre, String apellido, String correo) {
        try {
            URL url = new URL(URL_AUTOR);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("POST");
            conn.setDoOutput(true);
            // Parámetros basados en tu archivo crud.php: id, nombre, apellido, correo
            String params = "id=" + id
                    + "&nombre=" + nombre
                    + "&apellido=" + apellido
                    + "&correo=" + correo;
            OutputStream os = conn.getOutputStream();
            os.write(params.getBytes());
            os.flush();
            os.close();

            BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream(), "UTF-8"));
            StringBuilder response = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
            reader.close();
            return response.toString();
        } catch (Exception ex) {
            return ex.getMessage();
        }
    }

    // Método para buscar libros por título usando GET
    public static String buscarLibro(String titulo) {
        try {
            String urlString = URL_LIBRO + "?titulo=" + titulo;
            URL url = new URL(urlString);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream(), "UTF-8"));
            StringBuilder response = new StringBuilder();

    public void cargarTabla() {
        // Consumimos el método de libros del ApiService
        String json = ApiService.getLibros(); 
        Gson gson = new Gson();
        List<Libro> lista = gson.fromJson(
                json,
                new TypeToken<List<Libro>>() {}.getType());

        DefaultTableModel modelo = new DefaultTableModel();
        modelo.addColumn("ID Libro");
        modelo.addColumn("Título");
        modelo.addColumn("Año");
        modelo.addColumn("ID Autor");

        if (lista != null) {
            for (Libro l : lista) {
                modelo.addRow(new Object[]{
                    l.id, l.titulo, l.anio, l.autor_id
                });
            }
        }
        jTblLibros.setModel(modelo);
    }


    public void guardar() {
        String id = jTxtIdLibro.getText();
        String titulo = jTxtTitulo.getText();
        String anio = jTxtAnio.getText();
        String autorId = jTxtIdAutor.getText();

        // Se envía al API de Libros
        ApiService.insertarLibro(id, titulo, anio, autorId);
        cargarTabla();
    }

    public void borrar() {
        String id = jTxtIdLibro.getText();
        ApiService.borrarLibro(id);
        cargarTabla();
        inicial();
    }

    public void actualizar() {
        String id = jTxtIdLibro.getText();
        String titulo = jTxtTitulo.getText();
        String anio = jTxtAnio.getText();
        String autorId = jTxtIdAutor.getText();
        
        // Asumiendo que implementaste updateLibro en ApiService con esta estructura
        ApiService.updateLibro(id, titulo, anio, autorId); 
        cargarTabla();
        inicial();
    }

    public void buscar() {
        String tituloBusqueda = jTxtBuscar.getText().trim();

        if (tituloBusqueda.isEmpty()) {
            JOptionPane.showMessageDialog(this, "Ingrese un título para filtrar.");
            return;
        }

        String json = ApiService.buscarLibro(tituloBusqueda);

        if (json != null && !json.isEmpty() && !json.equals("null")) {
            Gson gson = new Gson();
            List<Libro> lista = gson.fromJson(json, new TypeToken<List<Libro>>() {}.getType());

            if (lista != null && !lista.isEmpty()) {
                Libro l = lista.get(0);

                DefaultTableModel modelo = new DefaultTableModel();
                modelo.addColumn("ID Libro");
                modelo.addColumn("Título");
                modelo.addColumn("Año");
                modelo.addColumn("ID Autor");

                modelo.addRow(new Object[]{ l.id, l.titulo, l.anio, l.autor_id });
                jTblLibros.setModel(modelo);

                // Llenar campos
                jTxtIdLibro.setText(l.id);
                jTxtTitulo.setText(l.titulo);
                jTxtAnio.setText(l.anio);
                jTxtIdAutor.setText(l.autor_id);
                
                jBtnCancelar.setEnabled(true);
                habilitarBotones();
            } else {
                JOptionPane.showMessageDialog(this, "No se encontró el libro.");
            }
        }
    }

    private void jTblLibrosMouseClicked(java.awt.event.MouseEvent evt) {                                        
        int fila = jTblLibros.getSelectedRow();
        if (fila != -1) {
            jTxtIdLibro.setText(jTblLibros.getValueAt(fila, 0).toString());
            jTxtTitulo.setText(jTblLibros.getValueAt(fila, 1).toString());
            jTxtAnio.setText(jTblLibros.getValueAt(fila, 2).toString());
            jTxtIdAutor.setText(jTblLibros.getValueAt(fila, 3).toString());
        }
    }
